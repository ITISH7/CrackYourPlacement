Given an array of strings words and a width maxWidth, format the text such that each line has exactly maxWidth characters and is fully (left and right) justified.

You should pack your words in a greedy approach; that is, pack as many words as you can in each line. Pad extra spaces ' ' when necessary so that each line has exactly maxWidth characters.

Extra spaces between words should be distributed as evenly as possible. If the number of spaces on a line does not divide evenly between words, the empty slots on the left will be assigned more spaces than the slots on the right.

For the last line of text, it should be left-justified, and no extra space is inserted between words.

Note:

A word is defined as a character sequence consisting of non-space characters only.
Each word's length is guaranteed to be greater than 0 and not exceed maxWidth.
The input array words contains at least one word.
 

Example 1:

Input: words = ["This", "is", "an", "example", "of", "text", "justification."], maxWidth = 16
Output:
[
   "This    is    an",
   "example  of text",
   "justification.  "
]
Example 2:

Input: words = ["What","must","be","acknowledgment","shall","be"], maxWidth = 16
Output:
[
  "What   must   be",
  "acknowledgment  ",
  "shall be        "
]
Explanation: Note that the last line is "shall be    " instead of "shall     be", because the last line must be left-justified instead of fully-justified.
Note that the second line is also left-justified because it contains only one word.
Example 3:

Input: words = ["Science","is","what","we","understand","well","enough","to","explain","to","a","computer.","Art","is","everything","else","we","do"], maxWidth = 20
Output:
[
  "Science  is  what we",
  "understand      well",
  "enough to explain to",
  "a  computer.  Art is",
  "everything  else  we",
  "do                  "
]
 

Constraints:

1 <= words.length <= 300
1 <= words[i].length <= 20
words[i] consists of only English letters and symbols.
1 <= maxWidth <= 100
words[i].length <= maxWidth

Solution :-

class Solution {
public:
    string createline(int curr_word_index , int last_word_index ,int number_of_spaces, int extraspace,vector<string>& words,int maxWidth){
        string ans ="";
        for(int i= curr_word_index;i<last_word_index;i++){
            ans+=words[i];
            if(i==last_word_index-1){
                continue;
            }
            for(int j = 0 ;j<number_of_spaces;j++){
                ans+=" ";
            }
            if(extraspace>0){
                ans+=" ";
                extraspace--;
            }
        }
        while(ans.size()<maxWidth){
            ans+=" ";
        }
        return ans;
    }
    vector<string> fullJustify(vector<string>& words, int maxWidth) {
        int curr_word_index = 0;
        vector<string> ans;
        int size = words.size();
        while (curr_word_index < size) {
            int next_word_index = curr_word_index + 1;
            int lettercount = words[curr_word_index].size();
            int number_of_spaces_between_words=0;
            while (next_word_index < size &&
                   (lettercount + 1 + words[next_word_index].size() + number_of_spaces_between_words <=
                    maxWidth)) {
                lettercount += words[next_word_index].size();
                number_of_spaces_between_words+=1;
                next_word_index++;
            }
            int total_space_left_in_line = maxWidth - lettercount;
            int number_of_blanks_required = number_of_spaces_between_words;
            int number_of_space_required_in_each_blanks = number_of_blanks_required==0?0: 
                total_space_left_in_line / number_of_blanks_required;
            int extraspaces_tobe_added =number_of_blanks_required==0?0:
                total_space_left_in_line % number_of_blanks_required;

                if(next_word_index==size){
                    //if last line
                    number_of_space_required_in_each_blanks =1;
                    extraspaces_tobe_added = 0;
                }

            ans.push_back(createline(curr_word_index,next_word_index,number_of_space_required_in_each_blanks, extraspaces_tobe_added , words,maxWidth));
            

            curr_word_index = next_word_index;
        }

        return ans;
    }
};
